# now we want to perform a regression to find K_d with an outlier detector scheme

# first we define a log posterior
# our log posterior is 


def f_efficiency(K_d, alpha, c_a_0, c_b_0):
    '''
    define efficiency
    '''
    
    #define what is in the sqrt of the denominator
    denom_sqrt = (K_d + c_a_0 + c_b_0)**2 - 4 * c_a_0 * c_b_0
    
    return 2 * alpha * c_b_0 / (K_d + c_a_0 + c_b_0 + np.sqrt(denom_sqrt))
                                
def resid(K_d, alpha, c_a_0, c_b_0, e):
    '''
    Residual for theorhetical spindle model.
    '''
    return e - f_efficiency(K_d, alpha, c_a_0, c_b_0)

def log_post(p, c_a_0, c_b_0, e):
    '''
    perform regressions to find K_d
    '''    
    # unpack parameters
    K_d, alpha = p
    
    # set boundaries
    if (alpha <= 0) or (alpha >= 1) or (K_d > 0):
        return -np.inf
    
    # define the log posterior
    return - len(c_a_0) / 2 * np.log(np.sum(resid(K_d, alpha, c_a_0, c_b_0, e)**2))

Now we want to preform a regression of $K_d$ without outliers. We do this by using a performing a cauchy distribution. We define a new log posterior.

In [25]:
def f_efficiency(K_d, alpha, c_a_0, c_b_0):
    '''
    define efficiency
    '''
    
    #define what is in the sqrt of the denominator
    denom_sqrt = (K_d + c_a_0 + c_b_0)**2 - 4 * c_a_0 * c_b_0
    
    return 2 * alpha * c_b_0 / (K_d + c_a_0 + c_b_0 + np.sqrt(denom_sqrt))
                                
                           
def resid(K_d, alpha, beta, c_a_0, c_b_0, e):
    '''
    Residual for theorhetical spindle model.
    '''
    
    return (e - f_efficiency(K_d, alpha, c_a_0, c_b_0)) / beta

def log_post_cauchy(p, c_a_0, c_b_0, e):
    '''
    perform regressions to find K_d
    '''
    
    K_d, alpha, beta = p

    if (alpha <= 0) or (alpha > 1) or (K_d <= 0) or (beta <= 0):
        return -np.inf
    
    # define the log posterior
    return - (len(c_a_0) + 1) * np.log(beta) - np.sum(np.log(1 + resid(K_d, alpha, beta, c_a_0, c_b_0, e)**2))
